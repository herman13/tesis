
\newcommand{\boxplotTiempos}[3]{

\begin{axis}[
  boxplot/draw direction=y,
  xtick={1, 2, 3},
  xticklabels={PDM, WDMM, VMM},
  ymin=0,
  ylabel={Tiempo de ejecución (horas)},
  xlabel={Modelo de movilidad},
]


\pgfplotstableread{#1}\pdm;
\pgfplotstableread{#2}\wdmm;
\pgfplotstableread{#3}\vmm;

\addplot+ [boxplot] table[y index = 0]{\pdm};
\addplot+ [boxplot] table[y index = 0]{\wdmm};
\addplot+ [boxplot] table[y index = 0]{\vmm};

\end{axis}
\end{tikzpicture}
}


\newcommand{\tablaTiempos}[2]{
\pgfplotstabletypeset[
  columns/A/.style={
    column name=#2
  },
  1000 sep={},
  columns/B/.style={column name=Tiempo (horas)},
  every even row/.style={
  before row={\rowcolor[gray]{0.9}}},
  every head row/.style={ before row=\toprule,after row=\midrule},
  every last row/.style={ after row=\bottomrule},
]{#1}
}
